-- Phase 4: Slug alias tracking for permanent redirects (D-13)
create table public.document_slug_aliases (
  id uuid primary key default gen_random_uuid(),
  kind text not null check (kind in ('invoice', 'quotation')),
  old_slug text not null,
  document_id uuid not null,
  user_id uuid not null references auth.users(id) on delete cascade,
  created_at timestamptz not null default now()
);

create index idx_slug_aliases_lookup on public.document_slug_aliases (kind, old_slug);

-- RLS: only the owner can read their own aliases
alter table public.document_slug_aliases enable row level security;

create policy "Users can read own slug aliases"
  on public.document_slug_aliases for select
  using (auth.uid() = user_id);

create policy "Users can insert own slug aliases"
  on public.document_slug_aliases for insert
  with check (auth.uid() = user_id);
```

/** UUID v4 detection for slug-vs-id route migration (D-12) */
export function isUuid(value: string): boolean {
  return /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i.test(value);
}

/** Format TRN for display with label prefix (SET-03, D-10) */
export function formatTrnDisplay(trn: string | null | undefined): string {
  if (!trn || trn.trim() === "") return "";
  return `TRN: ${trn.trim()}`;
}

/** Get Arabic description with English fallback for bilingual docs (SET-04) */
export function getArabicDescription(item: { description: string; arabicDescription?: string }): string {
  return item.arabicDescription && item.arabicDescription.trim() !== "" ? item.arabicDescription : item.description;
}
```

**Function 1: `acceptQuotationPublicAction`**

Signature: `export async function acceptQuotationPublicAction(shareToken: string, prevState: ActionState, formData: FormData): Promise<ActionState>`

Logic:
1. `const supabase = createSupabaseAdminClient();` — if null, return `{ status: "error", message: "Server configuration error." }`.
2. Query quotation: `.from("quotations").select("id, status").eq("share_token", shareToken).maybeSingle()`.
3. If no quotation: return `{ status: "error", message: "This link is no longer valid or has expired." }`.
4. If `quotation.status !== "sent"`: return `{ status: "error", message: "This quotation can no longer be accepted." }`. This prevents double-accept (Pitfall 2).
5. Update: `.from("quotations").update({ status: "accepted", accepted_date: new Date().toISOString() }).eq("id", quotation.id)`.
6. `revalidatePath(\`/quotations/public/${shareToken}\`);`
7. Return `{ status: "success", message: "Quotation accepted — the sender has been notified." }`.

**Function 2: `rejectQuotationPublicAction`**

Signature: `export async function rejectQuotationPublicAction(shareToken: string, prevState: ActionState, formData: FormData): Promise<ActionState>`

Logic:
1. Same admin client setup and null check.
2. Same quotation lookup by shareToken.
3. Same token validation and status guard (`status !== "sent"` returns error `"This quotation can no longer be rejected."`).
4. Extract `rejectionReason` from `formData.get("rejectionReason")` — cast to string, default to empty string `""` if null.
5. Update: `.from("quotations").update({ status: "rejected", rejected_date: new Date().toISOString(), rejection_reason: rejectionReason }).eq("id", quotation.id)`.
6. `revalidatePath(\`/quotations/public/${shareToken}\`);`
7. Return `{ status: "success", message: "Quotation rejected. Your note has been sent." }`.
