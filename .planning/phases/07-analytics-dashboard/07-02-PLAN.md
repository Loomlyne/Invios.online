```tsx
<div role="img" aria-label="Receivables aging breakdown by days outstanding">
  <ResponsiveContainer width="100%" height={120}>
    <BarChart data={chartData} layout="vertical" barSize={32}>
      <XAxis type="number" hide />
      <YAxis type="category" dataKey="name" hide />
      <Tooltip
        formatter={(value: number, name: string) => {
          const bucket = buckets.find((b) => b.label === name);
          return [`${formatCurrency(value as number, currency)} (${bucket?.count ?? 0} inv)`, name];
        }}
        contentStyle={{ borderRadius: 8, border: "1px solid rgba(28,25,23,0.12)", background: "#fffdf9" }}
      />
      {buckets.map((bucket, idx) => (
        <Bar key={bucket.label} dataKey={bucket.label} stackId="aging" fill={AGING_FILLS[idx]} radius={idx === 0 ? [3, 0, 0, 3] : idx === 3 ? [0, 3, 3, 0] : [0, 0, 0, 0]}>
        </Bar>
      ))}
    </BarChart>
  </ResponsiveContainer>
</div>
```
