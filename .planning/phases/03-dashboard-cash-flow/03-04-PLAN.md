```typescript
import { cn } from "@/lib/utils";

export function MetricCard({
  label,
  value,
  accent = false,
}: {
  label: string;
  value: string;
  accent?: boolean;
}) {
  return (
    <div className="rounded-[1.1rem] border border-black/7 bg-surface p-4 sm:p-5">
      <p className="text-xs text-muted uppercase tracking-[0.18em]">{label}</p>
      <p
        className={cn(
          "mt-1 text-2xl font-semibold tracking-tight",
          accent ? "text-success" : "text-foreground",
        )}
      >
        {value}
      </p>
    </div>
  );
}
```

Props:
- `label`: metric label text (e.g. "Total billed")
- `value`: formatted string (e.g. "AED 12,500" or "84%" or "---")
- `accent`: when true, value uses `text-success` (used for collection rate at 100%)

**Data loading:**
```typescript
export default async function AppHomePage() {
  const context = await getAppContext();
  const [metrics, recentInvoices, recentQuotations, overdueInvoices] = await Promise.all([
    getDashboardMetrics(context.userId),
    listRecentInvoices(context.userId, 5),
    listRecentQuotations(context.userId, 5),
    listOverdueInvoices(context.userId),
  ]);

  const currency = context.userState.settings.defaultCurrency;
  const hasData = recentInvoices.length > 0 || recentQuotations.length > 0;
  const setupItems = context.setupProgress.items;
  const nextItem = setupItems.find((item) => !item.complete) ?? setupItems[setupItems.length - 1];
```
