Reset function clears all state when dialog closes:
```typescript
function reset() {
  setStep(1);
  setRawRows([]);
  setRawHeaders([]);
  setMapping({});
  setValidatedRows([]);
  setResult(null);
}
function handleOpenChange(next: boolean) {
  setOpen(next);
  if (!next) reset();
}
```
