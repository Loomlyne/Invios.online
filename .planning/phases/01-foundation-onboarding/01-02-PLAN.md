**File 2: src/components/app/mobile-tab-bar.tsx — Create the component**

Create a client component with `"use client"` directive. The tab bar must:

1. Be a `<nav>` element with `fixed bottom-0 inset-x-0 z-30 lg:hidden` (hidden on desktop where sidebar exists).
2. Use a glass-effect style consistent with the UI spec: `border-t border-black/8 bg-white/92 backdrop-blur-md`.
3. Include `pb-[env(safe-area-inset-bottom)]` for iOS safe area (notch devices).
4. Render 5 tab items from `mobileTabItems` constant.
5. Use `usePathname()` from `next/navigation` for active state detection (reuse the `isActivePath` logic from `app-sidebar-nav.tsx`).
6. The "New Invoice" primary tab (where `primary: true`) should be visually distinct: use `bg-accent text-[#1C1917] rounded-full size-12 -mt-3 shadow-[0_8px_24px_rgba(202,138,4,0.28)]` for a raised FAB-style circle.
7. Import icons from `lucide-react`: `ReceiptText`, `UsersRound`, `Plus`, `FileText`, `Settings2`.
8. Active (non-primary) tabs use `text-accent-strong` for icon and `text-foreground font-medium` for label.
9. Inactive tabs use `text-muted` for both icon and label.
10. Each tab renders an icon (`size-5` for regular, `size-6` for primary) and a label (`text-[10px]` underneath, hidden for primary tab on small screens).
11. Height of the bar: `h-16` content area + safe area padding.

Implementation:

```typescript
"use client";

import Link from "next/link";
import { usePathname } from "next/navigation";
import { FileText, Plus, ReceiptText, Settings2, UsersRound } from "lucide-react";
import { mobileTabItems } from "@/lib/constants";
import { cn } from "@/lib/utils";

const iconMap: Record<string, typeof Plus> = {
  "receipt-text": ReceiptText,
  "users-round": UsersRound,
  "plus": Plus,
  "file-text": FileText,
  "settings-2": Settings2,
};

function isActivePath(pathname: string, href: string) {
  if (href === "/app") return pathname === "/app";
  return pathname === href || pathname.startsWith(`${href}/`);
}

export function MobileTabBar() {
  const pathname = usePathname();

  return (
    <nav className="fixed bottom-0 inset-x-0 z-30 lg:hidden border-t border-black/8 bg-white/92 backdrop-blur-md"
         style={{ paddingBottom: "env(safe-area-inset-bottom)" }}>
      <div className="flex items-end justify-around h-16 px-2">
        {mobileTabItems.map((item) => {
          const Icon = iconMap[item.icon];
          const active = isActivePath(pathname, item.href);

          if (item.primary) {
            return (
              <Link
                key={item.key}
                href={item.href}
                className="flex flex-col items-center justify-center -mt-3"
              >
                <span className="flex items-center justify-center size-12 rounded-full bg-accent text-[#1C1917] shadow-[0_8px_24px_rgba(202,138,4,0.28)] transition hover:-translate-y-0.5">
                  <Icon className="size-6" />
                </span>
                <span className="mt-1 text-[10px] font-medium text-accent-strong">{item.label}</span>
              </Link>
            );
          }

          return (
            <Link
              key={item.key}
              href={item.href}
              className="flex flex-col items-center justify-center gap-1 px-2 py-2"
            >
              <Icon className={cn("size-5", active ? "text-accent-strong" : "text-muted")} />
              <span className={cn(
                "text-[10px]",
                active ? "font-medium text-foreground" : "text-muted"
              )}>
                {item.label}
              </span>
            </Link>
          );
        })}
      </div>
    </nav>
  );
}
```
